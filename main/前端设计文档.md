## 一、项目概述

### 1.1 项目背景

基于 RuoYi-Cloud SpringBoot3 微服务架构，为教师和学生用户提供在线教育门户服务。系统支持视频点播、试题练习等核心功能，实现前后端完全分离架构。

### 1.2 用户角色

| 角色类型 | 说明 | 主要功能 |
|---------|------|---------|
| 教师 | edu_teacher 表用户 | 视频管理、试题管理、学生学习跟踪 |
| 学生 | edu_student 表用户 | 视频学习、试题练习、收藏管理 |
| 游客 | 未登录用户 | 浏览公开内容 |

### 2.3 渲染策略

| 页面类型 | 渲染方式 | 说明 |
|---------|---------|------|
| 首页 | SSR | 服务端渲染，SEO 友好 |
| 课程列表 | SSR | 服务端渲染，便于搜索引擎收录 |
| 课程详情 | SSR + CSR | 首屏 SSR，交互 CSR |
| 视频播放 | CSR | 客户端渲染，减少服务器压力 |
| 试题答题 | CSR | 客户端渲染，保护题库安全 |
| 个人中心 | CSR | 需要登录，客户端渲染 |

#### 4.1.2 视频播放器组件设计

```vue
<!-- components/video/VideoPlayer.vue -->
<template>
  <div class="video-player-wrapper">
    <video
      ref="videoRef"
      class="video-js vjs-big-play-centered"
      playsinline
    >
      <source :src="videoSrc" :type="videoType" />
    </video>
    
    <div v-if="showProgress" class="progress-bar">
      <span>上次观看: {{ formatTime(lastPosition) }}</span>
      <el-progress :percentage="progress" />
    </div>
  </div>
</template>

<script setup lang="ts">
import videojs from 'video.js'
import 'video.js/dist/video-js.css'

interface Props {
  videoSrc: string
  videoType?: string
  poster?: string
  autoplay?: boolean
  lastPosition?: number
}

interface Emits {
  (e: 'timeupdate', time: number): void
  (e: 'ended'): void
  (e: 'error', error: any): void
}

const props = withDefaults(defineProps<Props>(), {
  videoType: 'video/mp4',
  autoplay: false,
  lastPosition: 0
})

const emit = defineEmits<Emits>()

let player: any = null
const videoRef = ref<HTMLVideoElement>()
const progress = computed(() => {
  if (!player || !props.duration) return 0
  return Math.round((player.currentTime() / props.duration) * 100)
})

onMounted(() => {
  initPlayer()
})

onUnmounted(() => {
  if (player) {
    player.dispose()
  }
})

function initPlayer() {
  player = videojs(videoRef.value!, {
    controls: true,
    autoplay: props.autoplay,
    preload: 'auto',
    fluid: true,
    poster: props.poster,
    playbackRates: [0.5, 1, 1.25, 1.5, 2],
    controlBar: {
      children: [
        'playToggle',
        'volumePanel',
        'currentTimeDisplay',
        'timeDivider',
        'durationDisplay',
        'progressControl',
        'playbackRateMenuButton',
        'fullscreenToggle'
      ]
    }
  })
  
  if (props.lastPosition > 0) {
    player.currentTime(props.lastPosition)
  }
  
  player.on('timeupdate', () => {
    emit('timeupdate', player.currentTime())
  })
  
  player.on('ended', () => {
    emit('ended')
  })
}
</script>
```

```vue
<template>
  <div class="question-card" :class="{ 'is-correct': showResult && isCorrect }">
    <div class="question-header">
      <span class="question-type">{{ questionTypeText }}</span>
      <span class="question-difficulty">
        <el-rate
          v-model="difficulty"
          :max="5"

    </div>
    
    <div class="question-content" v-html="question.questionHtml" />
    
    <div v-if="question.questionImage" class="question-image">
      <el-image :src="question.questionImage" fit="contain" />
    </div>
    
    <div class="question-options">
      <QuestionOptions
        v-model="selectedAnswer"
        :options="question.options"
        :question-type="question.questionType"
        :show-result="showResult"
        :correct-answer="question.correctAnswer"
        :disabled="disabled"
      />
    </div>
    
    <div v-if="showResult" class="question-analysis">
      <div class="analysis-header">
        <span class="result-tag" :class="isCorrect ? 'correct' : 'wrong'">
          {{ isCorrect ? '回答正确' : '回答错误' }}
        </span>
      </div>
      <div class="analysis-content" v-html="question.analysisHtml" />
    </div>
    
    <div class="question-footer">
      <div class="question-meta">
        <span>知识点: {{ question.knowledgeName }}</span>
        <span>正确率: {{ question.correctRate }}%</span>
      </div>
      <div class="question-actions">
        <el-button v-if="!showResult" type="primary" @click="handleSubmit">
          提交答案
        </el-button>
        <el-button v-else @click="handleNext">下一题</el-button>
      </div>
    </div>
  </div>
</template>

<script setup lang="ts">
import type { Question } from '@/types/question'

interface Props {
  question: Question
  showResult?: boolean
  disabled?: boolean
}

interface Emits {
  (e: 'submit', answer: string): void
  (e: 'next'): void
}

const props = withDefaults(defineProps<Props>(), {
  showResult: false,
  disabled: false
})

const emit = defineEmits<Emits>()

const selectedAnswer = ref('')

const questionTypeText = computed(() => {
  const typeMap = {
    1: '单选题',
    2: '多选题',
    3: '判断题',
    4: '填空题',
    5: '简答题'
  }
  return typeMap[props.question.questionType]
})

const isCorrect = computed(() => {
  return selectedAnswer.value === props.question.correctAnswer
})

function handleSubmit() {
  emit('submit', selectedAnswer.value)
}

function handleNext() {
  emit('next')
}
</script>
```

```typescript
// utils/request.ts
import axios, { type AxiosInstance, type AxiosRequestConfig, type AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/stores/user'
import { getToken, removeToken } from '@/utils/auth'

const service: AxiosInstance = axios.create({
  baseURL: '/edu-user',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json'
  }
})

service.interceptors.request.use(
  (config) => {
    const token = getToken()
    if (token) {
      config.headers['Authorization'] = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

service.interceptors.response.use(
  (response: AxiosResponse) => {
    const res = response.data
    if (res.code !== 200) {
      ElMessage.error(res.msg || '请求失败')
      if (res.code === 401) {
        removeToken()
        window.location.href = '/login'
      }
      return Promise.reject(new Error(res.msg || '请求失败'))
    }
    return res
  },
  (error) => {
    ElMessage.error(error.message || '网络错误')
    return Promise.reject(error)
  }
)

export default service
```

```typescript
    practiceSession: null,

    currentIndex: 0,
    answers: {},
    results: {},
    elapsedTime: 0
  }),
  
  getters: {
    totalQuestions: (state) => state.questions.length,
    answeredCount: (state) => Object.keys(state.answers).length,
    correctCount: (state) => Object.values(state.results).filter(r => r.isCorrect).length,
    wrongCount: (state) => Object.values(state.results).filter(r => !r.isCorrect).length,
    correctRate: (state) => {
      const total = Object.keys(state.results).length
      if (total === 0) return 0
      return Math.round((Object.values(state.results).filter(r => r.isCorrect).length / total) * 100)
    }
  },
  
  actions: {
    async startPractice(params: { count?: number; knowledgeId?: number }) {
      const { startPractice } = useQuestionApi()
      const session = await startPractice(params)
      this.practiceSession = session
      this.questions = []
      this.currentIndex = 0
      this.answers = {}
      this.results = {}
      this.elapsedTime = 0
      return session
    },
    
    setQuestions(questions: Question[]) {
      this.questions = questions
      if (questions.length > 0) {
        this.currentQuestion = questions[0]
      }
    },
    
    setAnswer(questionId: number, answer: string) {
      this.answers[questionId] = answer
    },
    
    async submitAnswer(questionId: number) {
      const { submitAnswer } = useQuestionApi()
      const answer = this.answers[questionId]
      if (!answer) return null
      const result = await submitAnswer({ questionId, answer })
      this.results[questionId] = result
      return result
    },
    
    nextQuestion() {
      if (this.currentIndex < this.questions.length - 1) {
        this.currentIndex++
        this.currentQuestion = this.questions[this.currentIndex]
      }
    },
    
    prevQuestion() {
      if (this.currentIndex > 0) {
        this.currentIndex--
        this.currentQuestion = this.questions[this.currentIndex]
      }
    },
    
    jumpToQuestion(index: number) {
      if (index >= 0 && index < this.questions.length) {
        this.currentIndex = index
        this.currentQuestion = this.questions[index]
      }
    },
    
    updateElapsedTime(time: number) {
      this.elapsedTime = time
    },
    
    resetPractice() {
      this.practiceSession = null
      this.questions = []
      this.currentIndex = 0
      this.answers = {}
      this.results = {}
      this.elapsedTime = 0
      this.currentQuestion = null
```

```typescript
// composables/useCache.ts
export function useCache() {
  const cache = new Map<string, { data: any; expire: number }>()
  
  function get<T>(key: string): T | null {
    const item = cache.get(key)
    if (!item) return null
    if (Date.now() > item.expire) {
      cache.delete(key)
      return null
    }
    return item.data as T
  }
  
  function set(key: string, data: any, ttl: number = 5 * 60 * 1000) {
    cache.set(key, {
      data,
      expire: Date.now() + ttl
    })
  }
  
  function remove(key: string) {
    cache.delete(key)
  }
  
  function clear() {
    cache.clear()
  }
  
  return { get, set, remove, clear }
}
```

### 11.5 防抖节流

```typescript
// composables/useDebounce.ts
export function useDebounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number = 300
): T {
  let timer: ReturnType<typeof setTimeout> | null = null
  
  return ((...args: Parameters<T>) => {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn(...args)
    }, delay)
  }) as T
}

export function useThrottle<T extends (...args: any[]) => any>(
  fn: T,
  delay: number = 300
): T {
  let lastTime = 0
  
  return ((...args: Parameters<T>) => {
    const now = Date.now()
    if (now - lastTime >= delay) {
      fn(...args)
      lastTime = now
    }
  }) as T
}
```

```typescript
// stores/user.ts (扩展)
import { defineStore } from 'pinia'
import type { UserInfo } from '@/types'
import { getToken, setToken, removeToken, setUserType, getUserType } from '@/utils/auth'
import { useAuthApi } from '@/composables/useAuthApi'

interface UserState {
  token: string
  userInfo: UserInfo | null
  isLoggedIn: boolean
  userType: string
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    token: getToken() || '',
    userInfo: null,
    isLoggedIn: !!getToken(),
    userType: getUserType() || ''
  }),
  
  getters: {
    userId: (state) => state.userInfo?.userId || 0,
    userName: (state) => state.userInfo?.userName || '',
    realName: (state) => state.userInfo?.realName || '',
    avatar: (state) => state.userInfo?.avatar || '/images/default-avatar.png',
    phone: (state) => state.userInfo?.phone || '',
    email: (state) => state.userInfo?.email || '',
    teacherId: (state) => state.userInfo?.teacherId || (state.userInfo as any)?.teacherId,
    teacherNo: (state) => (state.userInfo as any)?.teacherNo || '',
    studentId: (state) => state.userInfo?.studentId || (state.userInfo as any)?.studentId,
    studentNo: (state) => (state.userInfo as any)?.studentNo || '',
    gradeName: (state) => (state.userInfo as any)?.gradeName || '',
    className: (state) => (state.userInfo as any)?.className || '',
    createTime: (state) => state.userInfo?.createTime || '',
    isTeacher: (state) => state.userType === 'teacher',
    isStudent: (state) => state.userType === 'student'
  },
  
  actions: {
    async login(username: string, password: string, userType: string, remember: boolean = false) {
      const { login } = useAuthApi()
      const res: any = await login({ username, password, userType })
      const token = res?.access_token || res?.token
      const type = res?.user_type || userType
      this.token = token
      this.isLoggedIn = true
      this.userType = type
      setToken(token, remember)
      setUserType(type)
      return res
    },
    
    async fetchUserInfo() {
      const { getUserInfo } = useAuthApi()
      const info = await getUserInfo()
      this.userInfo = info
      return info
    },
    
    async logout() {
      try {
        const { logout } = useAuthApi()
        await logout()
      } catch (e) {
        console.error('Logout error:', e)
      } finally {
        this.token = ''
        this.userInfo = null
        this.isLoggedIn = false
        this.userType = ''
        removeToken()
      }
    },
    
    updateUserInfo(info: Partial<UserInfo>) {
      if (this.userInfo) {
        this.userInfo = { ...this.userInfo, ...info }
      }
    }
  },
  
  persist: {
    key: 'edu-user',
    storage: typeof window !== 'undefined' ? localStorage : undefined,
    paths: ['token', 'userInfo', 'isLoggedIn', 'userType']
  }
})
```
